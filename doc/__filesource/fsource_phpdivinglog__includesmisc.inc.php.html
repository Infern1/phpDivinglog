  231. function is__writable($path) {

  233. //NOTE: use a trailing slash for folders!!!

  329. /**
  330.  * DECtoDMS  Converts decimal longitude / latitude to DMS
  331.  * ( Degrees / minutes / seconds )
  332.  * This is the piece of code which may appear to
  333.  * be inefficient, but to avoid issues with floating
  334.  * point math we extract the integer part and the float
  335.  * part by using a string function.
  336.  * @param mixed $dec 
  337.  * @access public
  338.  * @return void 
  339.  */
  340. function DECtoDMS($dec)
  341. {
  342.     if ($dec == "") {
  343.         $dms = "";
  344.     } else {
  345.         $vars = explode(".",$dec);
  346.         $deg = $vars[0];
  347.         $tempma = "0.".$vars[1];
  348.  
  349.         $tempma = $tempma * 3600;
  350.         $min = floor($tempma / 60);
  351.         $sec = $tempma - ($min*60);
  352.  
  353.         $dms = $deg . '&#176;';
  354.         if (($min != 0) || ($sec != 0)) {
  355.             $dms .= ' ' . $min . '&#8242;';
  356.             if ($sec != 0) {
  357. //                format to xx.xx
  358.                 $s = number_format($sec,2);
  359. //                remove trailing zeroes
  360.                 $s = rtrim($s,"0");
  361. //                remove a trailing decimal point
  362.                 $s = rtrim($s,".");
  363.                 $dms .= ' ' . $s . '&#8243;';
  364.             }
  365.         }
  366.     }
  367.     return $dms;
  368. }
