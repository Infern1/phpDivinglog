  204.     function parseControl($control, $parameter)
  205.     {
  206.         switch ($control) {
  207.             // font table definition start
  208.             case "fonttbl":
  209.                 $this->flags["fonttbl"] = true; // signal fonttable control records they are allowed to behave as expected
  210.                 break; 
  211.             // define or set font
  212.             case "f":
  213.                 if ($this->flags["fonttbl"]) { // if its set, the fonttable definition is written to; else its read from
  214.                     $this->flags["fonttbl_current_write"] = $parameter;
  215.                 } else {
  216.                     $this->flags["fonttbl_current_read"] = $parameter;
  217.                 } 
  218.                 break;
  219.             case "fcharset": 
  220.                 // this is for preparing flushQueue; it then moves the Queue to $this->fonttable .. instead to formatted output
  221.                 $this->flags["fonttbl_want_fcharset"] = $parameter;
  222.                 break;
  223.             case "fs": 
  224.                 // sets the current fontsize; is used by stylesheets (which are therefore generated on the fly
  225.                 $this->flags["fontsize"] = $parameter;
  226.                 break; 
  227.             // handle alignment
  228.             case "qc":
  229.                 $this->flags["alignment"] = "center";
  230.                 break;
  231.             case "qr":
  232.                 $this->flags["alignment"] = "right";
  233.                 break; 
  234.             // reset paragraph settings ( only alignment)
  235.             case "pard":
  236.                 $this->flags["alignment"] = "";
  237.                 break; 
  238.             // define new paragraph (for now, thats a simple break in html)
  239.             case "par": 
  240.                 // begin new line
  241.                 $this->flags["beginparagraph"] = true;

  244.                 }

  349.     /*
  350.         Dispatch start/end of logical rtf groups
  351.         (not every output type needs it; merely debugging purpose)
  352.     */
  353.     function flushGroup($state)
  354.     {
  355.         if ($state == "open") {
  356.             /* push onto the stack */
  357.             array_push($this->stack, $this->flags);
  358.  
  359.             if ($this->wantXML) {
  360.                 $this->out .= "<group>";
  361.             } 
  362.         } 
  363.         if ($state == "close") {
  364.             /* pop from the stack */
  365.             $this->last_flags = $this->flags;
  366.             $this->flags = array_pop($this->stack);
  367.  
  368.             $this->flags["fonttbl_current_write"] = ""; // on group close, no more font definition will be written to this id 
  369.             // this is not really the right way to do it !
  370.             // of course a '}' not necessarily donates a fonttable end; a fonttable
  371.             // group at least *can* contain sub-groups
  372.             // therefore an stacked approach is heavily needed
  373.             $this->flags["fonttbl"] = false; // no matter what you do, if a group closes, its fonttbl definition is closed too
  374.             
  375.             if ($this->wantXML) {
  376.                 $this->out .= "</group>";
  377.             } 
  378.         } 
  379.     }

  395.     function checkHtmlSpanContent($command)
  396.     {
  397.         reset($this->fontmodifier_table);
  398.         while (list($rtf, $html) = each($this->fontmodifier_table)) {
  399.             if ($this->flags[$rtf] == true) {
  400.                 if ($command == "start") {
  401.                     $this->out .= "<" . $html . ">";
  402.                 } else {
  403.                     $this->out .= "</" . $html . ">";
  404.                 } 
  405.             } 
  406.         } 
  407.     }

  409.     /*
  410.         flush text in queue
  411.     */
  412.     function flushQueue()
  413.     {
  414.         if (strlen($this->queue)) {
  415.             // processing logic
  416.             if (ereg("^[0-9]+$", $this->flags["fonttbl_want_fcharset"])) {
  417.                 $this->fonttable[$this->flags["fonttbl_want_fcharset"]]["charset"] = $this->queue;
  418.                 $this->flags["fonttbl_want_fcharset"] = "";
  419.                 $this->queue = "";
  420.             } 
  421.             // output logic
  422.             if (strlen($this->queue)) {
  423.                 /*
  424.                     Everything which passes this is (or, at leat, *should*) be only outputted plaintext
  425.                     Thats why we can safely add the css-stylesheet when using wantHTML
  426.                 */
  427.                 if ($this->wantXML) {
  428.                     $this->out .= "<plain>" . $this->queue . "</plain>";
  429.                 } 
  430.  
  431.                 if ($this->wantHTML) {
  432.                     // only output html if a valid (for now, just numeric;) fonttable is given
  433.                     if (ereg("^[0-9]+$", $this->flags["fonttbl_current_read"])) {
  434.                         if ($this->flags["beginparagraph"] == true) {
  435.                             $this->flags["beginparagraph"] = false;
  436.                             $this->out .= "<div align=\"";
  437.                             switch ($this->flags["alignment"]) {
  438.                                 case "right":
  439.                                     $this->out .= "right";
  440.                                     break;
  441.                                 case "center":
  442.                                     $this->out .= "center";
  443.                                     break;
  444.                                 case "left":
  445.                                 default:
  446.                                     $this->out .= "left";
  447.                             } 
  448.                             $this->out .= "\">";
  449.                         }

  576.     function makeStyles()
  577.     {

  584.     } 
  585.  
  586.     /*
  587.         finally ..
  588.  
  589.         How this parser (is supposed) to work:
  590.         ======================================
  591.         This parse simple starts at the beginning of the rtf core stream, 
  592.         catches every controlling character {,} and \, automatically builds 
  593.         control words and control symbols during his livetime, trashes 
  594.         every other character into the plain text queue
  595.     */
  596.     function parse()
  597.     {
  598.         $this->parserInit();
  599.  
  600.         $i = 0;
  601.         $this->cw = false; // flag if control word is currently parsed
  602.         $this->cfirst = false; // first control character ?
  603.         $this->cword = ""; // last or current control word ( depends on $this->cw
  604.         
  605.         $this->queue = ""; // plain text data found during parsing
  606.         
  607.         $this->flushHead();
  608.  
  609.         while ($i < $this->len) {
  610.             switch ($this->rtf[$i]) {
  611.                 case "{":
  612.                     if ($this->cw) {
  613.                         $this->flushControl();
  614.                         $this->cw = false;
  615.                         $this->cfirst = false;
  616.                     } else {
  617.                         $this->flushQueue();
  618.                     } 
  619.                     $this->flushGroup("open");
  620.                     break;
  621.                 case "}":
  622.                     if ($this->cw) {
  623.                         $this->flushControl();
  624.                         $this->cw = false;
  625.                         $this->cfirst = false;
  626.                     } else {
  627.                         $this->flushQueue();
  628.                     } 
  629.                     $this->flushGroup("close");
  630.                     break;
  631.                 case "\\":
  632.                     if ($this->cfirst) { // catches '\\'
  633.                         $this->queue .= '\\';
  634.                         $this->cfirst = false;
  635.                         $this->cw = false;
  636.                         break;
  637.                     } 
  638.                     if ($this->cw) {
  639.                         $this->flushControl();
  640.                     } else {
  641.                         $this->flushQueue();
  642.                     } 
  643.                     $this->cw = true;
  644.                     $this->cfirst = true;
  645.                     $this->cword = "";
  646.                     break;
  647.                 default:
  648.                     if ((ord($this->rtf[$i]) == 10) || (ord($this->rtf[$i]) == 13)) break; // eat line breaks
  649.                     if ($this->cw) { // active control word ?
  650.                         /*
  651.                             Watch the RE: there's an optional space at the end which IS part of
  652.                             the control word (but actually its ignored by flushControl)
  653.                         */
  654.                         if (ereg("^[a-zA-Z0-9-]?$", $this->rtf[$i])) { // continue parsing
  655.                             $this->cword .= $this->rtf[$i];
  656.                             $this->cfirst = false;
  657.                         } else {
  658.                             /*
  659.                                 Control word could be a 'control symbol', like \~ or \* etc.
  660.                             */
  661.                             $specialmatch = false;
  662.                             if ($this->cfirst) {
  663.                                 if ($this->rtf[$i] == '\'') { // expect to get some special chars
  664.                                     $this->flushQueue();
  665.                                     $this->flushSpecial($this->rtf[$i + 1] . $this->rtf[$i + 2]);
  666.                                     $i += 2;
  667.                                     $specialmatch = true;
  668.                                     $this->cw = false;
  669.                                     $this->cfirst = false;
  670.                                     $this->cword = "";
  671.                                 } else
  672.                                 if (ereg("^[{}\*]$", $this->rtf[$i])) {
  673.                                     $this->flushComment("control symbols not yet handled");
  674.                                     $specialmatch = true;
  675.                                 } 
  676.                                 $this->cfirst = false;
  677.                             } else {
  678.                                 if ($this->rtf[$i] == ' ') { // space delimtes control words, so just discard it and flush the controlword
  679.                                     $this->cw = false;
  680.                                     $this->flushControl();
  681.                                     break;
  682.                                 } 
  683.                             } 
  684.                             if (! $specialmatch) {
  685.                                 $this->flushControl();
  686.                                 $this->cw = false;
  687.                                 $this->cfirst = false;
  688.                                 /*
  689.                                     The current character is a delimeter, but is NOT
  690.                                     part of the control word so we hop one step back
  691.                                     in the stream and process it again
  692.                                 */
  693.                                 $i--;
  694.                             } 
  695.                         } 
  696.                     } else {
  697.                         // < and > need translation before putting into queue when XML or HTML is wanted
  698.                         if (($this->wantHTML) || ($this->wantXML)) {
  699.                             switch ($this->rtf[$i]) {
  700.                                 case "<":
  701.                                     $this->queue .= "&lt;";
  702.                                     break;
  703.                                 case ">":
  704.                                     $this->queue .= "&gt;";
  705.                                     break;
  706.                                 default:
  707.                                     $this->queue .= $this->rtf[$i];
  708.                                     break;
  709.                             } 
  710.                         } else {
  711.                             $this->queue .= $this->rtf[$i];
  712.                         } 
  713.                     } 
  714.             } 
  715.             $i++;
  716.         } 
  717.         $this->flushQueue();
  718.         $this->flushErrors();
  719.         $this->flushBottom();
  720.  
  721.         if ($this->wantHTML) {
  722.             $this->makeStyles();
  723.         } 
  724.         echo "<hr>\n<b>RTF Out</b><br>\n<code>\n";
  725.         echo "--->" . $this->out . "<---<br>\n";
  726.         echo "</code>\n<br>\n<hr>\n";
  727.     } 
  728. }
